// Function to display the info box for a star

function displayInfo(infoBox, star, event) {

    infoBox.style.display = "block";  // Show the info box

    let infoHTML = "";

    // Display information based on checkboxes

    if (showNameCheckbox.checked) {

        infoHTML += `<p><strong>Name:</strong> ${star.id}</p>`;

    }

    if (showLuminosityCheckbox.checked) {

        infoHTML += `<p><strong>Luminosity:</strong> ${star.lum} L\u2609</p>`;

    }

    if (showTemperatureCheckbox.checked) {

        infoHTML += `<p><strong>Temperature:</strong> ${star.temp}K</p>`;

    }

    if (showMassCheckbox.checked) {

        infoHTML += `<p><strong>Mass:</strong> ${star.mass} M\u2609</p>`;

    }

    if (showRadiusCheckbox.checked) {

        infoHTML += `<p><strong>Radius:</strong> ${star.rad} R\u2609</p>`;

    }

    if (showStageCheckbox.checked) {

        infoHTML += `<p><strong>Stage:</strong> ${star.stage}</p>`;

    }

    if (showLifetimeCheckbox.checked) {

        infoHTML += `<p><strong>Lifetime:</strong> ${star.life}</p>`;

    }

    if (showDistanceCheckbox.checked) {

        infoHTML += `<p><strong>Distance:</strong> ${star.dist} ly</p>`;

    }

    if (showElementsCheckbox.checked) {

        infoHTML += `<p><strong>Elements:</strong> ${star.elem}</p>`;

    }

    if (showImageCheckbox.checked) {

        infoHTML += `${star.image}</p>`;

    }

    infoBox.innerHTML = infoHTML;  // Update the info box content

    moveInfoBox(infoBox, event);  // Move the info box next to the mouse

}

// Function to move the info box to the mouse cursor position

function moveInfoBox(infoBox, event) {

    const containerRect = container.getBoundingClientRect(); // Get the position of the container

    const offsetX = 15; // A little padding from the cursor

    const offsetY = 15; // A little padding from the cursor

    const mouseX = event.clientX - containerRect.left + offsetX;

    const mouseY = event.clientY - containerRect.top + offsetY;

    infoBox.style.left = `${mouseX}px`;

    infoBox.style.top = `${mouseY}px`;

}

// Function to hide the info box

function hideInfo(infoBox) {

    infoBox.style.display = "none";  // Hide the info box

}

// Function to update the information when checkboxes are changed

function updateInfo() {

    const infoBoxes = document.querySelectorAll('.info-box');

    infoBoxes.forEach(infoBox => {

        const starId = infoBox.id.split('-')[1];  // Extract the star ID from the info-box ID

        const star = stars.find(s => s.id === starId);

        if (infoBox.style.display === "block") {

            displayInfo(infoBox, star);  // Recalculate the info if the box is visible

        }

    });

}
